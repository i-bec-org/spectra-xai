spectraxai.dataset

The module spectraxai.dataset includes classes and operations on datasets.

This module defines the class Dataset which ought to be used to handle all necessary dataset transformations including scaling, spectral pre-processing, and splitting. The classes defined in spectraxai.models and spectraxai.explain generally expect as inputs a Dataset object to operate on.

 63class Dataset:
 64    """A general class to manage the dataset (i.e. input X and output Y).
 65
 66    Use this class to pass your 2D spectral matrix and 1D or 2D output
 67    properties. Supports methods for pre-processing X, scaling X and Y,
 68    splitting the dataset, and more.
 69
 70    """
 71
 72    n_samples: int
 73    """Number of samples in the dataset"""
 74    n_features: int
 75    """Number of features in the dataset's input"""
 76    n_outputs: int
 77    """Number of outputs in the dataset's output"""
 78    X: np.ndarray
 79    """The spectral data of size (`n_samples`, `n_features`)"""
 80    Y: np.ndarray
 81    """The output properties of size (`n_samples`, `n_outputs`)"""
 82    X_names: List[str]
 83    """The names of the input features of size (`n_features`)"""
 84    Y_names: List[str]
 85    """The names of the output properties of size (`n_outputs`)"""

 87    def __init__(
 88        self,
 89        X: np.ndarray,
 90        Y: np.ndarray,
 91        X_names: List[str] = [],
 92        Y_names: List[str] = [],
 93    ):
 94        """Create a new Dataset from input and output arrays.
 95
 96        Parameters
 97        ----------
 98        X: `numpy.ndarray`
 99            A 2D matrix of the spectra of size (`n_samples`, `n_features`)
100
101        Y: `numpy.ndarray`
102            A 1D vector (`n_samples`,) or 2D matrix (`n_samples`, `n_outputs`)
103            of the output property(ies). If 1D it will be implicitly converted
104            to 2D.
105
106        X_names: `list[str]`, optional
107            A list of length `n_features` containing the names of the input
108            features. If missing, these will be autogenerated as X1, X2, etc.
109
110        Y_names: `list[str]`, optional
111            A list of length `n_outputs` containing the names of the output
112            properties. When using a single output property pass a list of
113            size 1. If missing, these will be autogenerated as Y1, Y2, etc.
114
115        """
116        if X.shape[0] != Y.shape[0]:
117            raise AssertionError("X and Y don't have the same number of rows!")
118        if X.ndim != 2:
119            raise AssertionError("X should have exactly two dimensions")
120        self.X = X.to_numpy() if isinstance(X, pandas.DataFrame) else X
121        if isinstance(Y, pandas.DataFrame) or isinstance(Y, pandas.Series):
122            Y = Y.to_numpy()
123        self.Y = Y if Y.ndim > 1 else Y.reshape(-1, 1)
124        check_X_y(X, Y, multi_output=True)
125        self.n_samples = self.X.shape[0]
126        self.n_features = self.X.shape[1]
127        self.n_outputs = self.Y.shape[1]
128        if X_names and len(X_names) != self.n_features:
129            raise AssertionError("X_names should have the same n_features as X")
130        if Y_names and len(Y_names) != self.n_outputs:
131            raise AssertionError("Y_names should have the same n_outputs as Y")
132        self.X_names = (
133            X_names if X_names else ["X" + str(i + 1) for i in range(self.n_features)]
134        )
135        self.Y_names = (
136            Y_names if Y_names else ["Y" + str(i + 1) for i in range(self.n_outputs)]
137        )

139    def train_test_split(self, split: DatasetSplit, opt: Number) -> DataSplit:
140        """Split dataset with passed split method to train and test.
141
142        Parameters
143        ----------
144        split: `DatasetSplit`
145            The method used to split the dataset
146
147        opt: `Number`
148            A float number (between 0 and 1) indicating the percentage of the
149            training dataset for Random and Kennard–Stone split. A natural number
150            for Cross Validation and Stratified split.
151
152        Returns
153        -------
154        `DataSplit`
155            The idx_trn, idx_tst tuple. Use e.g. X[idx_trn], Y[idx_trn] to get the
156            training dataset.
157
158        """
159        indices = np.arange(self.X.shape[0])
160        if split == DatasetSplit.RANDOM or split == DatasetSplit.KENNARD_STONE:
161            msg = (
162                "opt param should be a float in the (0, 1) range for "
163                "RANDOM and KENNARD_STONE splits"
164            )
165            if not isinstance(opt, float):
166                raise TypeError(msg)
167            if opt <= 0 or opt >= 1:
168                raise ValueError(msg)
169        elif split == DatasetSplit.CROSS_VALIDATION or split == DatasetSplit.STRATIFIED:
170            msg = (
171                "opt param should be a positive int greater than 1 for "
172                "CROSS_VALIDATION and STRATIFIED splits"
173            )
174            if not isinstance(opt, int):
175                raise TypeError(msg)
176            if opt <= 1:
177                raise ValueError(msg)
178        if split == DatasetSplit.RANDOM:
179            return train_test_split(indices, train_size=opt)
180        elif split == DatasetSplit.KENNARD_STONE:
181            _, _, idx_trn, idx_tst = kennard_stone.train_test_split(
182                self.X, indices, test_size=(1 - opt)
183            )
184            return idx_trn, idx_tst
185        elif split == DatasetSplit.CLHS:
186            raise NotImplementedError("clhs not implemented yet")
187        elif split == DatasetSplit.CROSS_VALIDATION:
188            kf = KFold(opt, shuffle=True)
189            idx_trn, idx_tst = [], []
190            for trn_index, tst_index in kf.split(self.X):
191                idx_trn.append(np.array(trn_index))
192                idx_tst.append(np.array(tst_index))
193            return np.array(idx_trn, dtype=object), np.array(idx_tst, dtype=object)
194        elif split == DatasetSplit.STRATIFIED:
195            skf = StratifiedKFold(n_splits=opt)
196            idx_trn, idx_tst = [], []
197            for trn_index, tst_index in skf.split(self.X, self.Y):
198                idx_trn.append(np.array(trn_index))
199                idx_tst.append(np.array(tst_index))
200            return np.array(idx_trn, dtype=object), np.array(idx_tst, dtype=object)
201        else:
202            raise RuntimeError("Not a valid split method!")

224    def train_test_split_explicit(
225        self, trn: np.ndarray = np.array([]), tst: np.ndarray = np.array([])
226    ) -> DataSplit:
227        """Split dataset to train and test from pre-selected by the user indices.
228
229        Parameters
230        ----------
231        trn: `np.ndarray`, optional
232            Contains the indices of the training samples
233
234        tst: `np.ndarray`, optional
235            Contains the indices of the testing samples
236
237        Returns
238        -------
239        `DataSplit`
240            The idx_trn, idx_tst tuple
241
242        """
243        if tst.size == 0 and trn.size == 0:
244            raise AssertionError("You need to specify either tst or trn indices")
245        if tst.size > 0 and trn.size > 0:
246            raise AssertionError("You cannot specify both trn and tst")
247        if tst.size > 0 and isinstance(tst[0], np.ndarray):
248            trn = np.array(
249                [self.train_test_split_explicit(tst=fold)[0] for fold in tst]
250            )
251        elif tst.size > 0:
252            if not np.logical_and(tst >= 0, tst <= self.n_samples).all():
253                raise AssertionError("Passed indices contain out of bound values")
254            trn = np.array(list(set(range(0, self.n_samples)).difference(set(tst))))
255        elif trn.size > 0 and isinstance(trn[0], np.ndarray):
256            tst = np.array(
257                [self.train_test_split_explicit(trn=fold)[1] for fold in trn]
258            )
259        else:
260            if not np.logical_and(trn >= 0, trn <= self.n_samples).all():
261                raise AssertionError("Passed indices contain out of bound values")
262            tst = np.array(list(set(range(0, self.n_samples)).difference(set(trn))))
263        return trn, tst

358    def unscale_X(
359        X: np.ndarray, method: Scale, set_params: List = [], set_attributes: List = []
360    ):
361        """Unscale X matrix of the spectra with Scale method.
362
363        Parameters
364        ----------
365        X: `numpy.ndarray`
366            A 2D or 3D scaled matrix of the spectra.
367
368        method: `Scale`
369            The method is used to scale X
370
371        set_params: `List`
372            A list of dicts with the parameters of each Scale method.
373
374        set_attributes: `List`
375            A list of dicts with the attributes of each Scale method.
376
377
378        Returns
379        -------
380        `numpy.ndarray`
381            The original X matrix of the spectra.
382
383        """
384        if len(set_attributes) == 0:
385            raise AssertionError("You need to specify set_attributes")
386        if X.ndim not in [2, 3]:
387            raise AssertionError("X must be either a 2-D or a 3-D matrix")
388        if X.ndim == 3:
389            if method == Scale.STANDARD:
390                scaler = [
391                    [StandardScaler() for _ in range(X.shape[1])]
392                    for _ in range(X.shape[2])
393                ]
394            elif method == Scale.MINMAX:
395                scaler = [
396                    [MinMaxScaler() for _ in range(X.shape[1])]
397                    for _ in range(X.shape[2])
398                ]
399            for i in range(X.shape[2]):
400                X[:, :, i] = Dataset.__unscale_X_parser(
401                    X[:, :, i], method, scaler[i], set_params[i], set_attributes[i]
402                )
403        else:
404            if method == Scale.STANDARD:
405                scaler = [StandardScaler() for _ in range(X.shape[1])]
406            elif method == Scale.MINMAX:
407                scaler = [MinMaxScaler() for _ in range(X.shape[1])]
408            X = Dataset.__unscale_X_parser(
409                X, method, scaler, set_params[0], set_attributes[0]
410            )
411        return X
