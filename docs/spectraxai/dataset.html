spectraxai.dataset

The module spectraxai.dataset includes classes and operations on datasets.

This module defines the class Dataset which ought to be used to handle all necessary dataset transformations including scaling, spectral pre-processing, and splitting. The classes defined in spectraxai.models and spectraxai.explain generally expect as inputs a Dataset object to operate on.

203class Dataset:
204    """A general class to manage the dataset (i.e. input X and output Y).
205
206    Use this class to pass your 2D spectral matrix and 1D or 2D output
207    properties. Supports methods for pre-processing X, scaling X and Y,
208    splitting the dataset, and more.
209
210    """
211
212    n_samples: int
213    """Number of samples in the dataset."""
214    n_features: int
215    """Number of features in the dataset's input."""
216    n_outputs: int
217    """Number of outputs in the dataset's output."""
218    X: np.ndarray
219    """The spectral data of size (`n_samples`, `n_features`)"""
220    Y: np.ndarray
221    """The output properties of size (`n_samples`, `n_outputs`)"""
222    X_names: List[str]
223    """The names of the input features of size (`n_features`)"""
224    Y_names: List[str]
225    """The names of the output properties of size (`n_outputs`)"""
226
227    def __init__(
228        self,
229        X: np.ndarray,
230        Y: np.ndarray,
231        X_names: List[str] = [],
232        Y_names: List[str] = [],
233    ):
234        """Create a new Dataset from input and output arrays.
235
236        Parameters
237        ----------
238        X: `numpy.ndarray` or `pandas.DataFrame`
239            A 2D matrix of the spectra of size (`n_samples`, `n_features`)
240
241        Y: `numpy.ndarray` or `pandas.DataFrame` or `pandas.Series`
242            A 1D vector (`n_samples`,) or 2D matrix (`n_samples`, `n_outputs`)
243            of the output property(ies). If 1D it will be implicitly converted
244            to 2D.
245
246        X_names: `list[str]`, optional
247            A list of length `n_features` containing the names of the input
248            features. If missing, these will be autogenerated as X1, X2, etc.
249
250        Y_names: `list[str]`, optional
251            A list of length `n_outputs` containing the names of the output
252            properties. When using a single output property pass a list of
253            size 1. If missing, these will be autogenerated as Y1, Y2, etc.
254
255        """
256        if X.shape[0] != Y.shape[0]:
257            raise AssertionError("X and Y don't have the same number of rows!")
258        if X.ndim != 2:
259            raise AssertionError("X should have exactly two dimensions")
260        self.X = X.to_numpy() if isinstance(X, pandas.DataFrame) else X
261        if isinstance(Y, pandas.DataFrame) or isinstance(Y, pandas.Series):
262            Y = Y.to_numpy()
263        self.Y = Y if Y.ndim > 1 else Y.reshape(-1, 1)
264        check_X_y(X, Y, multi_output=True)
265        self.n_samples = self.X.shape[0]
266        self.n_features = self.X.shape[1]
267        self.n_outputs = self.Y.shape[1]
268        if X_names and len(X_names) != self.n_features:
269            raise AssertionError("X_names should have the same n_features as X")
270        if Y_names and len(Y_names) != self.n_outputs:
271            raise AssertionError("Y_names should have the same n_outputs as Y")
272        self.X_names = (
273            X_names if X_names else ["X" + str(i + 1) for i in range(self.n_features)]
274        )
275        self.Y_names = (
276            Y_names if Y_names else ["Y" + str(i + 1) for i in range(self.n_outputs)]
277        )
278
279    def _repair_split_array(self, arr):
280        arr = np.array(arr, dtype=object)
281        if arr[0].dtype == np.dtype("O"):
282            arr = arr.astype(np.int64)
283        return arr

285    def train_test_split(self, split: DatasetSplit, opt: Number) -> DataSplit:
286        """Split dataset with passed split method to train and test.
287
288        Parameters
289        ----------
290        split: `DatasetSplit`
291            The method used to split the dataset
292
293        opt: `Number`
294            A float number (between 0 and 1) indicating the percentage of the
295            training dataset for Random and KennardStone split. A natural number
296            for Cross Validation and Stratified split.
297
298        Returns
299        -------
300        `DataSplit`
301            The idx_trn, idx_tst tuple. Use e.g. X[idx_trn], Y[idx_trn] to get the
302            training dataset.
303
304        """
305        indices = np.arange(self.X.shape[0])
306        if split == DatasetSplit.RANDOM or split == DatasetSplit.KENNARD_STONE:
307            msg = (
308                "opt param should be a float in the (0, 1) range for "
309                "RANDOM and KENNARD_STONE splits"
310            )
311            if not isinstance(opt, float):
312                raise TypeError(msg)
313            if opt <= 0 or opt >= 1:
314                raise ValueError(msg)
315        elif split == DatasetSplit.CROSS_VALIDATION or split == DatasetSplit.STRATIFIED:
316            msg = (
317                "opt param should be a positive int greater than 1 for "
318                "CROSS_VALIDATION and STRATIFIED splits"
319            )
320            if not isinstance(opt, int):
321                raise TypeError(msg)
322            if opt <= 1:
323                raise ValueError(msg)
324        if split == DatasetSplit.RANDOM:
325            return train_test_split(indices, train_size=opt)
326        elif split == DatasetSplit.KENNARD_STONE:
327            _, _, idx_trn, idx_tst = kennard_stone.train_test_split(
328                self.X, indices, test_size=(1 - opt)
329            )
330            return idx_trn, idx_tst
331        elif split == DatasetSplit.CLHS:
332            raise NotImplementedError("clhs not implemented yet")
333        elif split == DatasetSplit.CROSS_VALIDATION:
334            kf = KFold(opt, shuffle=True)
335            idx_trn, idx_tst = [], []
336            for trn_index, tst_index in kf.split(self.X):
337                idx_trn.append(np.array(trn_index))
338                idx_tst.append(np.array(tst_index))
339            return self._repair_split_array(idx_trn), self._repair_split_array(idx_tst)
340        elif split == DatasetSplit.STRATIFIED:
341            skf = StratifiedKFold(n_splits=opt)
342            idx_trn, idx_tst = [], []
343            for trn_index, tst_index in skf.split(self.X, self.Y):
344                idx_trn.append(np.array(trn_index))
345                idx_tst.append(np.array(tst_index))
346            return self._repair_split_array(idx_trn), self._repair_split_array(idx_tst)
347        else:
348            raise RuntimeError("Not a valid split method!")
349
350    def train_test_split_explicit(
351        self, trn: np.ndarray = np.array([]), tst: np.ndarray = np.array([])
352    ) -> DataSplit:
353        """Split dataset to train and test from pre-selected by the user indices.
354
355        This is useful in cases where one has the set containing the training
356        indices of a dataset and wants to get the complement of the training set to
357        obtain the test set.
358
359        Parameters
360        ----------
361        trn: `np.ndarray`, optional
362            Contains the indices of the training samples
363
364        tst: `np.ndarray`, optional
365            Contains the indices of the testing samples
366
367        Returns
368        -------
369        `DataSplit`
370            The idx_trn, idx_tst tuple
371
372        """
373        if tst.size == 0 and trn.size == 0:
374            raise AssertionError("You need to specify either tst or trn indices")
375        if tst.size > 0 and trn.size > 0:
376            raise AssertionError("You cannot specify both trn and tst")
377        if tst.size > 0 and isinstance(tst[0], np.ndarray):
378            trn = [self.train_test_split_explicit(tst=fold)[0] for fold in tst]
379        elif tst.size > 0:
380            if not np.logical_and(tst >= 0, tst <= self.n_samples).all():
381                raise AssertionError("Passed indices contain out of bound values")
382            trn = list(set(range(0, self.n_samples)).difference(set(tst)))
383        elif trn.size > 0 and isinstance(trn[0], np.ndarray):
384            tst = [self.train_test_split_explicit(trn=fold)[1] for fold in trn]
385        else:
386            if not np.logical_and(trn >= 0, trn <= self.n_samples).all():
387                raise AssertionError("Passed indices contain out of bound values")
388            tst = list(set(range(0, self.n_samples)).difference(set(trn)))
389        trn, tst = np.array(trn), np.array(tst)
390        if isinstance(trn[0], np.ndarray) and trn[0].dtype == "O":
391            trn = trn.astype(np.int64)
392        if isinstance(tst[0], np.ndarray) and tst[0].dtype == "O":
393            tst = tst.astype(np.int64)
394        return trn, tst
