 55    def __init__(
 56        self,
 57        model: Model,
 58        init_hyperparameters: Dict = {},
 59        grid_search_hyperparameters: Dict = {},
 60    ):
 61        """
 62        Parameters
 63        ----------
 64
 65        model: `Model`
 66            Select a model from `Model` class.
 67
 68        init_hyperparameters: `dict`, optional
 69            A dictionary of pre-selected hyperparameters (e.g. a best model)
 70
 71        grid_search_hyperparameters: `dict`, optional
 72            Specify custom grid search range for the hyperparameters
 73        """
 74        if len(init_hyperparameters) != 0 and len(grid_search_hyperparameters) != 0:
 75            raise AssertionError(
 76                "You cannot specify both init_hyperparameters and grid_search_hyperparameters"
 77            )
 78        self.model = model
 79        self.init_hyperparameters = init_hyperparameters
 80        if len(grid_search_hyperparameters) != 0:
 81            self.grid_search_hyperparameters = grid_search_hyperparameters
 82        else:
 83            if model == Model.SVR:
 84                self.grid_search_hyperparameters = {
 85                    "epsilon": [0.01, 0.025, 0.05, 0.1, 0.15, 0.2, 0.5],
 86                    "C": np.logspace(start=-2, stop=9, base=2, num=12),
 87                }
 88            elif model == Model.RF:
 89                self.grid_search_hyperparameters = {
 90                    "max_features": ["auto", "sqrt", "log2"],
 91                    "n_estimators": [50, 100, 150, 200],
 92                }
 93            elif model == Model.PLS:
 94                # Not set because it depends on the dataset's numbers of features
 95                self.grid_search_hyperparameters = {}
 96            elif model == Model.CUBIST:
 97                self.grid_search_hyperparameters = {
 98                    "n_committees": [1, 5, 10, 20],
 99                    "neighbors": [1, 5, 9],
100                    "composite": [True],
101                }
102        self.training_time = None
103        self.testing_time = None
104        self.best_model = None
105        self.best_score = None

107    def __vip(self, model):
108        t = model.x_scores_
109        w = model.x_weights_
110        q = model.y_loadings_
111        p, h = w.shape
112        vips = np.zeros((p,))
113        s = np.diag(t.T @ t @ q.T @ q).reshape(h, -1)
114        total_s = np.sum(s)
115        for i in range(p):
116            weight = np.array(
117                [(w[i, j] / np.linalg.norm(w[:, j])) ** 2 for j in range(h)]
118            )
119            vips[i] = np.sqrt(p * (s.T @ weight) / total_s)
120        return vips

122    def train(self, dataset: Dataset, cv: Union[int, List] = 5):
123        """Trains the model on a given dataset.
124
125        If you didn't supply the `init_hyperparameters` option to the constructor, then
126        a grid search optimization process takes place as follows:
127        Using the sklearn.model_selection.GridSearchCV approach, a grid search using a
128        cross-validation splitting strategy specified by cv is performed. After the optimal
129        hyperparameters are defined, the model is then retrained on the whole dataset.
130
131        Parameters
132        ----------
133
134        dataset: `spectraxai.dataset.Dataset`
135            the Dataset to train the model
136
137        cv: int, or an iterable, default=5
138            Determines the cross-validation splitting strategy.
139            Possible inputs for cv are:
140            * integer, to specify the number of folds
141            * An iterable yielding (train, test) splits as arrays of indices
142        """
143        if self.model in [Model.SVR, Model.CUBIST] and dataset.n_outputs > 1:
144            raise AssertionError(
145                "Cannot create a multi-output SVR model. Please train a model for each output property."
146            )
147
148        # Scale the data for SVR
149        if self.model == Model.SVR:
150            self.scaler = StandardScaler()
151            self.scaler.fit(dataset.X)
152            X_train = self.scaler.transform(dataset.X)
153        else:
154            X_train = dataset.X
155
156        # If hyperparameters are passed, use them; otherwise run grid search
157        if len(self.init_hyperparameters) != 0:
158            if self.model == Model.SVR:
159                self.best_model = SVR(kernel="rbf", max_iter=5e8)
160                self.best_model = self.best_model.set_params(
161                    **self.init_hyperparameters
162                )
163            elif self.model == Model.PLS:
164                self.best_model = PLSRegression()
165                self.best_model = self.best_model.set_params(
166                    **self.init_hyperparameters
167                )
168            elif self.model == Model.RF:
169                self.best_model = RandomForestRegressor()
170                self.best_model = self.best_model.set_params(
171                    **self.init_hyperparameters
172                )
173            elif self.model == Model.CUBIST:
174                self.best_model = Cubist(**self.init_hyperparameters)
175            trn_t0 = time.time()
176            self.best_model.fit(X_train, np.squeeze(dataset.Y))
177            self.best_hyperparameters = self.init_hyperparameters
178            trn_t1 = time.time()
179        else:
180            if self.model == Model.SVR:
181                if "gamma" not in self.grid_search_hyperparameters:
182                    gamma = sigest(X_train)
183                    self.grid_search_hyperparameters["gamma"] = np.linspace(
184                        gamma[0], gamma[2], num=5
185                    )
186                model = SVR(kernel="rbf", max_iter=5e8)
187            elif self.model == Model.PLS:
188                if "n_components" not in self.grid_search_hyperparameters:
189                    self.grid_search_hyperparameters["n_components"] = np.arange(
190                        1, min(100, X_train.shape[1]), 1
191                    )
192                model = PLSRegression()
193            elif self.model == Model.RF:
194                model = RandomForestRegressor()
195            elif self.model == Model.CUBIST:
196                model = Cubist()
197            trn_t0 = time.time()
198            grid_search = GridSearchCV(
199                estimator=model,
200                param_grid=self.grid_search_hyperparameters,
201                cv=cv,
202                n_jobs=-1,
203                verbose=0,
204            )
205            grid_search.fit(X_train, np.squeeze(dataset.Y))
206            trn_t1 = time.time()
207            self.best_model = grid_search.best_estimator_
208            self.best_hyperparameters = grid_search.best_params_
209            self.best_score = grid_search.best_score_
210        self.training_time = trn_t1 - trn_t0

212    def predict(self, X_test: np.ndarray) -> np.ndarray:
213        """Predict using the best_model from a new unknown input
214
215        Parameters
216        ----------
217        X_test: `np.ndarray`
218            The new input data to predict their output
219
220        Returns
221        -------
222        `np.ndarray`
223            A np.ndarray of size (n_test_samples, n_outputs) with the predictions
224        """
225        if self.best_model is None:
226            raise RuntimeError("You need to train the model first")
227        if X_test.ndim == 1:
228            X_test = X_test.reshape(1, -1)
229        tst_t0 = time.time()
230        if self.model == Model.SVR:
231            X_test = self.scaler.transform(X_test)
232        y_hat = self.best_model.predict(X_test)
233        tst_t1 = time.time()
234        self.testing_time = tst_t1 - tst_t0
235        return y_hat

237    def train_and_test(
238        self,
239        dataset: Dataset,
240        preprocess: SpectralPreprocessingSequence = SpectralPreprocessing.NONE,
241        idx_trn: np.ndarray = np.array([]),
242        idx_tst: np.ndarray = np.array([]),
243        get_model: bool = False,
244    ) -> List[Dict]:
245        """Trains and tests a model given a dataset and a spectral pre-processing sequence.
246
247        Pass here the whole dataset of (X, Y) and either specify idx_trn (training indices) or idx_tst (testing indices).
248
249        Parameters
250        ----------
251
252        dataset: `spectraxai.dataset.Dataset`
253            the Dataset to train the model
254
255        preprocess: `spectraxai.spectra.SpectralPreprocessingSequence`
256            Optional pre-processing sequence. Defaults to SpectralPreprocessing.NONE.
257
258        idx_trn: `np.array`
259            The indices of the trn samples. Defaults to np.array([]).
260
261        idx_tst: `np.array`
262            The indices of the tst samples. Defaults to np.array([]).
263
264        get_model: `bool`
265            If true, also return the generated model. Defaults to False.
266
267        Returns
268        -------
269        `List[Dict]`
270            A dictionary containing the accuracy results and assorted metadata for each output property
271        """
272
273        if idx_tst.size == 0 and idx_trn.size == 0:
274            raise AssertionError(
275                "You need to specify either tst or trn indices or both"
276            )
277        if idx_tst.size > 0 and idx_trn.size > 0:
278            raise AssertionError("You cannot specify both trn and tst")
279        elif idx_trn.size > 0 and idx_tst.size == 0:
280            idx_trn, idx_tst = dataset.train_test_split_explicit(trn=idx_trn)
281        elif idx_trn.size == 0 and idx_tst.size > 0:
282            idx_trn, idx_tst = dataset.train_test_split_explicit(tst=idx_tst)
283
284        dataset = dataset.preprocess(preprocess)
285        self.train(Dataset(dataset.X[idx_trn], dataset.Y[idx_trn]))
286        y_hat = self.predict(dataset.X[idx_tst])
287
288        results = []
289        for i in range(dataset.n_outputs):
290            res = metrics(
291                dataset.Y[idx_tst, i], y_hat[:, i] if y_hat.ndim == 2 else y_hat
292            )
293            res["output"] = dataset.Y_names[i]
294            # Gather all results
295            res["pre_process"] = str(preprocess)
296            res["val_score"] = self.best_score
297            if self.model == Model.SVR:
298                for key in ["C", "epsilon", "gamma"]:
299                    res[key] = self.best_hyperparameters[key]
300                res["SVs"] = len(self.best_model.support_)
301            elif self.model == Model.PLS:
302                res["n_components"] = self.best_hyperparameters["n_components"]
303                res["VIP"] = self.__vip(self.best_model)
304            elif self.model == Model.RF:
305                for key in ["max_features", "n_estimators"]:
306                    res[key] = self.best_hyperparameters[key]
307                res["feature_importance"] = self.best_model.feature_importances_
308            elif self.model == Model.CUBIST:
309                for key in ["n_committees", "neighbors"]:
310                    res[key] = self.best_hyperparameters[key]
311                res["feature_importance"] = self.best_model.feature_importances_
312            res["training_time"] = self.training_time
313            res["testing_time"] = self.testing_time
314            if get_model:
315                res["model"] = self.best_model
316            results.append(res)
317        return results

319    def train_and_test_with_sequence(
320        self,
321        dataset: Dataset,
322        preprocesses: List[SpectralPreprocessingSequence] = [],
323        idx_trn: np.ndarray = np.array([]),
324        idx_tst: np.ndarray = np.array([]),
325    ) -> pandas.DataFrame:
326        """Train a model using different SpectralPreprocessingSequences and predict on the test set
327
328        A short-hand version to quickly test different pre-treatments, calling the train_and_test function
329
330        Parameters
331        ----------
332        dataset: `spectraxai.dataset.Dataset`
333            the Dataset to train the model
334
335        preprocesses: `spectraxai.spectra.List[SpectralPreprocessingSequence]`
336            List of different pre-processing sequences to test. Defaults to [].
337
338        idx_trn: `np.array`
339            The indices of the trn samples. Defaults to np.array([]).
340
341        idx_tst: `np.array`
342            The indices of the tst samples. Defaults to np.array([]).
343
344        Returns
345        -------
346        `pandas.DataFrame`
347            Returns a dataframe with the results of the trained models. By default, no model is returned to keep a low memory footprint.
348        """
349        if len(preprocesses) == 0:
350            raise ValueError(
351                "The list of SpectralPreprocessingSequence may not be empty"
352            )
353        if idx_tst.size == 0 and idx_trn.size == 0:
354            raise AssertionError("You need to specify either tst or trn indices")
355        if idx_tst.size > 0 and idx_trn.size > 0:
356            raise AssertionError("You cannot specify both trn and tst")
357        common = [
358            "pre_process",
359            "training_time",
360            "testing_time",
361            "val_score",
362            "N",
363            "RMSE",
364            "R2",
365            "RPIQ",
366        ]
367        if (idx_trn.size > 0 and isinstance(idx_trn[0], np.ndarray)) or (
368            idx_tst.size > 0 and isinstance(idx_tst[0], np.ndarray)
369        ):
370            common += ["fold"]
371        if self.model == Model.PLS:
372            dataFrame = pandas.DataFrame(columns=common + ["n_components", "VIP"])
373        elif self.model == Model.SVR:
374            dataFrame = pandas.DataFrame(
375                columns=common + ["epsilon", "C", "gamma", "SVs"]
376            )
377        elif self.model == Model.RF:
378            dataFrame = pandas.DataFrame(
379                columns=common + ["max_features", "n_estimators", "feature_importance"]
380            )
381        elif self.model == Model.CUBIST:
382            dataFrame = pandas.DataFrame(
383                columns=common + ["n_committees", "neighbors", "feature_importance"]
384            )
385        for preprocess in preprocesses:
386            if idx_trn.size > 0 and isinstance(idx_trn[0], np.ndarray):
387                for fold in range(idx_trn.shape[0]):
388                    for result in self.train_and_test(
389                        dataset, preprocess, idx_trn=idx_trn[fold]
390                    ):
391                        result["fold"] = fold + 1
392                        dataFrame.loc[len(dataFrame)] = result
393            elif idx_tst.size > 0 and isinstance(idx_tst[0], np.ndarray):
394                for fold in range(idx_tst.shape[0]):
395                    for result in self.train_and_test(
396                        dataset, preprocess, idx_tst=idx_tst[fold]
397                    ):
398                        result["fold"] = fold + 1
399                        dataFrame.loc[len(dataFrame)] = result
400            else:
401                for result in self.train_and_test(
402                    dataset, preprocess, idx_trn, idx_tst
403                ):
404                    dataFrame.loc[len(dataFrame)] = result
405        return dataFrame

388                for fold in range(idx_trn.shape[0]):
389                    for result in self.train_and_test(
390                        dataset, preprocess, idx_trn=idx_trn[fold]
391                    ):
392                        result["fold"] = fold + 1
393                        dataFrame.loc[len(dataFrame)] = result
394            elif idx_tst.size > 0 and isinstance(idx_tst[0], np.ndarray):
395                for fold in range(idx_tst.shape[0]):
396                    for result in self.train_and_test(
397                        dataset, preprocess, idx_tst=idx_tst[fold]
398                    ):
399                        result["fold"] = fold + 1
400                        dataFrame.loc[len(dataFrame)] = result
401            else:
402                for result in self.train_and_test(
403                    dataset, preprocess, idx_trn, idx_tst

405                    dataFrame.loc[len(dataFrame)] = result

Train a model using different SpectralPreprocessingSequences and predict on the test set

A short-hand version to quickly test different pre-treatments, calling the train_and_test function

Parameters
  • dataset (spectraxai.dataset.Dataset): the Dataset to train the model
  • preprocesses (spectraxai.spectra.List[SpectralPreprocessingSequence]): List of different pre-processing sequences to test. Defaults to [].
  • idx_trn (np.array): The indices of the trn samples. Defaults to np.array([]).
  • idx_tst (np.array): The indices of the tst samples. Defaults to np.array([]).
Returns
  • pandas.DataFrame: Returns a dataframe with the results of the trained models. By default, no model is returned to keep a low memory footprint.
